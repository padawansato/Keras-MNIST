Adadelta¶

.

\begin{align} r &\leftarrow \gamma r + (1 - \gamma) g_\vec{w}^{2} \\ v &\equiv \frac{\sqrt{s + \epsilon}}{\sqrt{r + \epsilon}} g_\vec{w} \\ w &\leftarrow w - \alpha v \\ s &\leftarrow \gamma s + (1 - \gamma) v^{2} \end{align}

RMSprop¶

\begin{align} r &\leftarrow \gamma r + (1 - \gamma) g_\vec{w}^{2} \\ w &\leftarrow w - \frac{\alpha}{\sqrt{r} + \epsilon} g_\vec{w} \end{align}

from keras.layers import Conv2D, MaxPooling2D
from keras import backend as K

batch_size = 128
num_classes = 10
epochs = 12

img_rows, img_cols = 28, 28 ##

(x_train, y_train), (x_test, y_test) = mnist.load_data()

#KerasTensorFlowTheano
if K.image_data_format() == 'channels_first':
    x_train = x_train.reshape(x_train.shape[0], 1, img_rows, img_cols)
    x_test = x_test.reshape(x_test.shape[0], 1, img_rows, img_cols)
    input_shape = (1, img_rows, img_cols)
else:
    x_train = x_train.reshape(x_train.shape[0], img_rows, img_cols, 1)
    x_test = x_test.reshape(x_test.shape[0], img_rows, img_cols, 1)
    input_shape = (img_rows, img_cols, 1)
